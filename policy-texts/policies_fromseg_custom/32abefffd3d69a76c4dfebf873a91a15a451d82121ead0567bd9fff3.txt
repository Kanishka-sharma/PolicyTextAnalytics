	function isNullOrWhitespace( input ) {
		if (input == null) return true;
		return input.replace(/\s/gi, '').
length < 1;
	}

	function setCookie(cookieName, cookieValue, expirydays) {
		var expireDate = new Date();
		expireDate.setTime(expireDate.getTime() + (expirydays * 24 * 60 * 60 * 1000));
		var expires = "expires=" + expireDate.toGMTString();
		document.cookie = cookieName + "=" + cookieValue + ";domain=channelsight.com;" + expires + ";path=/";
	}

	function getCookie(cookieName) {
		var name = cookieName + "=";
		var decodedCookie = decodeURIComponent(document.cookie);
		var cookieArray = decodedCookie.split(';');
		for(var i = 0; i < cookieArray.length; i++) {
			var cookiePair = cookieArray[i].
replace(/\s/gi, '');
			if (cookiePair.indexOf(name) == 0) {
				return cookiePair.substring(name.length, cookiePair.length);
			}
		}
		return "";
	}
